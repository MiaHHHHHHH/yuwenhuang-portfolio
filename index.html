  <script>

    // 1) 让滚轮 = 横向滚动（更像画廊）

    (function(){

      const row = document.getElementById('heroRow');

      if(!row) return;


      row.addEventListener('wheel', (e) => {

        const delta = Math.abs(e.deltaX) > Math.abs(e.deltaY) ? e.deltaX : e.deltaY;

        if (Math.abs(delta) < 2) return;

        e.preventDefault();

        row.scrollLeft += delta;

      }, { passive:false });

    })();


    // 2) 点击大图 -> 打开作品信息弹窗

    (function(){

      const modal = document.getElementById('modal');

      const closeBtn = document.getElementById('closeBtn');


      const mTitle = document.getElementById('mTitle');

      const mMetaTop = document.getElementById('mMeta');

      const mMetaLine = document.getElementById('mMetaLine');

      const mDesc = document.getElementById('mDesc');

      const mImg = document.getElementById('mImg');

      const mLink = document.getElementById('mLink');


      function openModal(data){

        mTitle.textContent = data.title || '';

        mMetaTop.textContent = data.meta || '';

        mMetaLine.textContent = data.meta || '';

        mDesc.textContent = data.desc || '';

        mImg.src = data.img || '';

        mImg.alt = data.title || '';


        if(data.link){

          mLink.style.display = 'inline-block';

          mLink.href = data.link;

        }else{

          mLink.style.display = 'none';

          mLink.href = '#';

        }


        modal.classList.add('show');

        modal.setAttribute('aria-hidden', 'false');

        document.body.style.overflow = 'hidden';

      }


      function closeModal(){

        modal.classList.remove('show');

        modal.setAttribute('aria-hidden', 'true');

        document.body.style.overflow = '';

      }


      document.querySelectorAll('.panelBtn').forEach(btn => {

        btn.addEventListener('click', () => {

          openModal({

            title: btn.dataset.title,

            meta: btn.dataset.meta,

            desc: btn.dataset.desc,

            link: btn.dataset.link,

            img: btn.dataset.img

          });

        });

      });


      closeBtn.addEventListener('click', closeModal);

      modal.addEventListener('click', (e) => {

        if(e.target === modal) closeModal();

      });

      window.addEventListener('keydown', (e) => {

        if(e.key === 'Escape' && modal.classList.contains('show')) closeModal();

      });

    })();


    // 3) 小头：随鼠标微微动（轻、丝滑）

    (function(){

      const wrap = document.getElementById('heads');

      if(!wrap) return;


      const prefersReduce = window.matchMedia && window.matchMedia("(prefers-reduced-motion: reduce)").matches;

      if(prefersReduce) return;


      const heads = Array.from(wrap.querySelectorAll('.head'));

      if(!heads.length) return;


      // 每个头不同 depth（越大动得越明显）

      const depths = {

        yellow: 10,

        orange: 14,

        cyan: 12,

        grey: 10,

        blue: 12,

        pink: 18

      };


      let targetX = 0, targetY = 0;

      let currentX = 0, currentY = 0;

      const ease = 0.08;


      function onMove(e){

        const r = wrap.getBoundingClientRect();

        const x = (e.clientX - r.left) / r.width;   // 0..1

        const y = (e.clientY - r.top) / r.height;   // 0..1

        targetX = (x - 0.5);

        targetY = (y - 0.5);

      }


      function tick(){

        currentX += (targetX - currentX) * ease;

        currentY += (targetY - currentY) * ease;


        heads.forEach(el => {

          const cls = el.className;

          let d = 12;

          if(cls.includes('yellow')) d = depths.yellow;

          else if(cls.includes('orange')) d = depths.orange;

          else if(cls.includes('cyan')) d = depths.cyan;

          else if(cls.includes('grey')) d = depths.grey;

          else if(cls.includes('blue')) d = depths.blue;

          else if(cls.includes('pink')) d = depths.pink;


          const tx = -currentX * d;

          const ty = -currentY * d;

          el.style.transform = `translate3d(${tx}px, ${ty}px, 0)`;

        });


        requestAnimationFrame(tick);

      }


      wrap.addEventListener('mousemove', onMove);

      wrap.addEventListener('mouseleave', () => { targetX = 0; targetY = 0; });

      tick();

    })();
